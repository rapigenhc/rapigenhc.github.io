const getTrafficSource = () => {
    const params = new URLSearchParams(window.location.search);
    return {
        utm_source: params.get('utm_source') || 'direct',
        utm_medium: params.get('utm_medium') || 'none',
        utm_campaign: params.get('utm_campaign') || 'none'
    };
};

window.submitForm = async function() {
    if (window.isSubmitting) return;

    // 도배 방지: localStorage 활용 (Spark 요금제 보호)
    const lastSubmit = localStorage.getItem('last_submit');
    if (lastSubmit && Date.now() - lastSubmit < 60000) {
        showAlert("잠시 후 다시 시도해주세요.");
        return;
    }

    const nameInput = document.getElementById('userName');
    const phoneInput = document.getElementById('userPhone');
    const agree = document.getElementById('agree');
    
    // 유효성 검사 (정규식 보안)
    if (!agree?.checked) { showAlert("개인정보 동의가 필요합니다."); return; }
    if (!/^[가-힣a-zA-Z\s]{2,20}$/.test(nameInput.value)) { showAlert("이름을 확인해주세요."); return; }
    if (!/^01[016789]\d{7,8}$/.test(phoneInput.value.replace(/-/g,''))) { showAlert("번호를 확인해주세요."); return; }

    // 단일 선택된 패키지 추출[cite: 10]
    const selectedRadio = document.querySelector('input[name="package"]:checked');
    const pkgName = selectedRadio ? selectedRadio.value : "미선택";

    try {
        window.isSubmitting = true;
        const source = getTrafficSource();

        // 1. Firestore 저장
        await addDoc(collection(db, "reservations"), {
            name: nameInput.value.trim(),
            phone: phoneInput.value.replace(/-/g,''),
            package: pkgName,
            status: "대기",
            source: source.utm_source,
            medium: source.utm_medium,
            createdAt: serverTimestamp()
        });

        // 2. GA4 이벤트 전송[cite: 10]
        if (typeof gtag === 'function') {
            gtag('event', 'conversion', { 'event_label': pkgName });
        }

        localStorage.setItem('last_submit', Date.now());
        document.getElementById('successModal')?.classList.remove('hidden');
        document.getElementById('successModal')?.classList.add('flex');

    } catch (e) {
        showAlert("오류가 발생했습니다.");
    } finally {
        window.isSubmitting = false;
    }
};
